    <script type="text/javascript">
        $(document).ready(function () {
            $("input:checkbox[data-agenda]").click(function (e) {
                $('#btnExcluirAgenda').prop("disabled", true);
                $('#btnExcluirAgenda').css("visibility", "hidden");
                $('#btnConfirmarPresenca').css("display", "none");
                $('#btnConfirmarAusencia').css("display", "none");
                $('#btnConfirmarPresenca').prop("disabled", true);
                $('#btnConfirmarAusencia').prop("disabled", true);

                var arr = $("input:checkbox[data-agenda]:checked");
                if (arr.length > 0) {

                    if ($("#IdStatus").val() != '6')
                    {
                        $('#btnExcluirAgenda').prop("disabled", false);
                        $('#btnExcluirAgenda').css("visibility", "visible");
                    }
                    
                    if ($("#IdStatus").val() == '6')
                    {
                        $('#btnConfirmarPresenca').prop("disabled", false);
                        $('#btnConfirmarPresenca').show();

                        $('#btnConfirmarAusencia').prop("disabled", false);
                        $('#btnConfirmarAusencia').show();
                    }
                    
                }
            });

            $('#btnExcluirAgenda').click(function (e) {
                var arr = $("input:checkbox[data-agenda]:checked");
                if (arr.length > 0) {
                    //Exibir modal
                    $('#myModalFiltroBody').html('Confirma a exclusão do(s) agendamento(s) selecionado(s)?');
                    $("#myModalFiltro").modal({ backdrop: 'static' });
                }
            });





            //$('.modal-backdrop').css('visibility', 'hidden'); pesquisar(this);
            //$('#divDetalhes').delegate('#btnModalFiltroConcluir', 'click', function (e) {
            //});
            $('#btnModalFiltroConcluir').click(function (e) {
                var removerAgd = '';
                var arr = $("input:checkbox[data-agenda]:checked");
                for (var i = 0; i < arr.length; i++) {
                    removerAgd = removerAgd + $(arr[i]).val() + '|';
                }
                if (removerAgd != '') {
                    var customerId = $('#CustomerId').val();
                    var userId = $('#UserId').val();
                    var idPrestador = $('#IdPrestador').val();

                    var acao = $("#ActExcluirAgendas").val();
                    $.ajax({
                        url: acao,
                        type: 'POST',
                        data: { CustomerId: customerId, UserId: userId, IdPrestador: idPrestador, Agendas: removerAgd },
                        success: function (response) {
                            var myObject = eval('(' + JSON.stringify(response) + ')');
                            if (myObject.Erro == '') {
                                //alert(myObject.Mensagem);
                                //Exibir modal
                                showModal("Sucesso", myObject.Mensagem);
                            }
                        }
                    });
                }

            });

        });


    </script>
